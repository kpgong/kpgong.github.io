#include <bits/stdc++.h>
using namespace std;
using C = complex<double>;

const double PI = acos(-1);

void FFT(vector<C>& x, bool inv)
{
    int n = x.size();
    if (n == 1) return;
    int h = n / 2;
    vector<C> odd(h), even(h);
    for (int i = 0; i < h; i++)
    {
        even[i] = x[i << 1];
        odd[i] = x[i << 1 | 1];
    }
    FFT(odd, inv);
    FFT(even, inv);

    double rad = inv ? PI / h : -PI / h;
    C acc = 1, w = exp(1.0i * rad), t;
    for (int k = 0; k < h; k++)
    {
        t = acc * odd[k];
        x[k] = (even[k] + t);
        x[k + h] = (even[k] - t);
        acc *= w;
    }
}

int extend(int a)
{
    int ret = 1;
    while (ret < a) ret <<= 1;
    return ret;
}

int main()
{
    int n, m, len;
    cin >> n >> m;
    len = extend(n + m + 1);
    vector<C> f(len), g(len);
    for (int i = 0; i <= n; i++) cin >> f[i];
    for (int i = 0; i <= m; i++) cin >> g[i];
    FFT(f, false);
    FFT(g, false);
    for (int i = 0; i < len; i++) f[i] *= g[i];
    FFT(f, true);
    for (int i = 0, e = n + m; i <= e; i++)
        cout << (int) (abs(f[i]) / len + .5) << ' ';
    cout << '\n';
    return 0;
}
