代码实现 FFT

函数 FFT 接受一组离散数序列 x，要求离散数序列 x 长度为 222 的正整数幂。inv 指示是否为逆运算。变换后的序列会放在 x 中。没有将结果作为返回是为了节约时间。

这段代码选用的规范化因子是 111。因为是递归实现的，所以取其他规范化因子会不便于实现。如果要同时用到正变换和逆变换，注意在调用 FFT 函数之后手动调整规范化因子——正反变换的规范化因子之积应为 1N\frac{1}{N}N1​。

using C = complex<double>;
void FFT(vector<C>& x, bool inv)
{
    int n = x.size();
    if (n == 1) return;
    vector<C> odd, even;
    for (int i = 0; i < n; i++)
    {
        if (i % 2 == 0) even.push_back(x[i]);
        else odd.push_back(x[i]);
    }
    FFT(odd, inv);
    FFT(even, inv);

    int h = n / 2;
    double ang = inv ? PI / h : -PI / h;
    C acc = 1, w = { cos(ang), sin(ang) };
    for (int k = 0; k < h; k++)
    {
        x[k] = even[k] + acc * odd[k];
        x[k + h] = even[k] - acc * odd[k];
        acc *= w;
    }
}
